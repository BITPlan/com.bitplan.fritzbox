135   /**
136    * handle the command line command
137    * @throws Throwable 
138    */
139   public void doCommand() throws Throwable {
140     FritzBoxSession session=FritzBoxSessionImpl.getInstance();
141     if (this.doGetCallList) {
142       final CallAPI callList = new CallAPIImpl(session);
143       List<Call> calls = callList.getCallList();
144       for (Call call:calls) {
145         System.out.println(call);
146       }
147     } else {
148       final HomeAutomation homeAutomation = new HomeAutomationImpl(session);
149 
150       final DeviceList deviceList = homeAutomation.getDeviceListInfos();
151       if (debug) {
152         LOGGER.log(Level.INFO,
153             String.format("Found %s devices", deviceList.devices.size()));
154         for (final Device device : deviceList.devices) {
155           LOGGER.log(Level.INFO, String.format("\t%s", device));
156         }
157       }
158 
159       final List<String> ids = homeAutomation.getSwitchList();
160       if (debug) {
161         LOGGER.log(Level.INFO,
162             String.format("Found %3d device ids: %s", ids.size(), ids));
163       }
164 
165       if (deviceList.devices.isEmpty()) {
166         session.logout();
167         return;
168       }
169 
170       if (listDevices) {
171         show("%20s | %10s | %25s | %s", "Name", "By", "Product", "Identifier");
172         show("%21s+%12s+%27s+%s", dash(21), dash(12), dash(27), dash(25));
173         for (final Device device : deviceList.devices) {
174           show("%20s | %10s | %25s | %s", device.name, device.manufacturer,
175               device.productname, device.identifier);
176         }
177       }
178       final Map<String, String> ainByName = new HashMap<>();
179       for (final Device device : deviceList.devices) {
180         ainByName.put(device.name, device.getAin());
181       }
182       /**
183        * loop over all devices to be read
184        */
185       for (final String readDevice : readDevices) {
186         // devices can be specified by name or id
187         String ain = readDevice;
188         // check if a name was given
189         if (ainByName.containsKey(ain)) {
190           // get the ain for the name
191           ain = ainByName.get(ain);
192         }
193         readSwitch(homeAutomation, ain);
194       }
195 
196       /**
197        * loop over all devices to be set
198        */
199       if (setDevices.size() % 2 != 0) {
200         usage("set needs pairs of name=on/off");
201       }
202       for (int i = 0; i < setDevices.size(); i += 2) {
203         final String name = setDevices.get(i);
204         final String powerState = setDevices.get(i + 1);
205         // check if a name was given
206         String ain = name;
207         if (ainByName.containsKey(ain)) {
208           // get the ain for the name
209           ain = ainByName.get(ain);
210         }
211         boolean newState = false;
212         switch (powerState) {
213         case "on":
214           newState = true;
215           break;
216         case "off":
217           newState = false;
218           break;
219         default:
220           usage(String.format(
221               "%s is not a valid powerState it needs to be on or off",
222               powerState));
223         }
224         show("switching %s %s", name, powerState);
225         homeAutomation.setSwitchOnOff(ain, newState);
226       }
227     }
228   }

230   /**
231    * read the given switch
232    * 
233    * @param homeAutomation
234    * @param ain
235    *          - the identification of the switch
236    * @throws Exception
237    */
238   private void readSwitch(HomeAutomation homeAutomation, String ain)
239       throws Exception {
240     show(" name: %s", homeAutomation.getSwitchName(ain));
241     show("   id: %s ", ain);
242     show("alive: %s", homeAutomation.getSwitchPresent(ain));
243     show("   on: %s", homeAutomation.getSwitchState(ain));
244     show(" uses: %5.0f     W", homeAutomation.getSwitchPowerWatt(ain));
245     show(" used: %9.3f kWh", homeAutomation.getSwitchEnergy(ain) / 1000.0);
246     show(" temp: %7.1f   °C", homeAutomation.getTemperature(ain));
247     show("");
248   }

260   /**
261    * get the given number of dashes
262    * 
263    * @param number
264    * @return the number of dashes
265    */
266   public String dash(int number) {
267     return s('-', number);
268   }
269 
270   /**
271    * get a string with number repetitions of the given char
272    * 
273    * @param c
274    * @param number
275    * @return - the repeated char
276    */
277   public String s(char c, int number) {
278     final StringBuffer sb = new StringBuffer();
279     for (int i = 0; i < number; i++) {
280       sb.append(c);
281     }
282     return sb.toString();
283   }

285   /**
286    * work on the given commands
287    * 
288    * @throws Exception
289    *           if a problem occurs
290    */
291   protected void work() throws Throwable {
292     if (this.showVersion || this.debug) {
293       showVersion();
294     }
295     if (this.showHelp) {
296       showHelp();
297     } else {
298       doCommand();
299     }
300   }
301 
302   /**
303    * main routine
304    * 
305    * @param args
306    *          - the command line arguments
307    * @return - the exit statusCode
308    * 
309    */
310   public int maininstance(String[] args) {
311     parser = new CmdLineParser(this);
312     try {
313       parser.parseArgument(args);
314       work();
315       exitCode = 0;
316     } catch (final CmdLineException e) {
317       // handling of wrong arguments
318       usage(e.getMessage());
319     } catch (final Throwable th) {
320       handle(th);
321       exitCode = 1;
322     }
323     return exitCode;
324   }
325 
326   /**
327    * main routine
328    * 
329    * @param args
330    *          - command line arguments
331    */
332   public static void main(String[] args) {
333     final CmdLine cmdLine = new CmdLine();
334     final int result = cmdLine.maininstance(args);
335     if (!testMode) {
336       System.exit(result);
337     }
338   }

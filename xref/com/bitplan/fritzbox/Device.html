21  package com.bitplan.fritzbox;
22  
23  import javax.xml.bind.annotation.XmlAttribute;
24  import javax.xml.bind.annotation.XmlElement;
25  import javax.xml.bind.annotation.XmlRootElement;
26  import javax.xml.bind.annotation.XmlTransient;
27  
28  @XmlRootElement(name="device")
29  public class Device {
30    @XmlAttribute
31    public String identifier;
32    @XmlAttribute
33    public String id;
34    @XmlAttribute
35    public int functionbitmask;
36    @XmlAttribute
37    public String fwversion;
38    @XmlAttribute
39    public String manufacturer;
40    @XmlAttribute
41    public String productname;
42    @XmlElement
43    public int present;
44    @XmlElement
45    public String name;
46    @XmlElement(name="switch")
47    public SwitchState state;
48    @XmlElement
49    PowerMeter powermeter;
50    @XmlElement
51    Temperature temperature;
52    
53    @XmlTransient
54    public String getAin() {
55      String ain=identifier.replace(" ", "");
56      return ain;
57    }
58    
59    @Override
60    public String toString() {
61      return "Device [identifier=" + identifier + ", id=" + id
62          + ", functionBitmask=" + functionbitmask + ", firmwareVersion="
63          + fwversion + ", manufacturer=" + manufacturer + ", productName="
64          + productname + ", present=" + present + ", name=" + name
65          + ", switchState=" + state + ", powerMeter=" + powermeter
66          + ", temperature=" + temperature + "]";
67    }
68  
69    
70  }

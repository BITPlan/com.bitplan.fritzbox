21  package com.bitplan.fritzbox;
22  
23  import java.util.List;
24  import java.util.logging.Level;
25  import java.util.logging.Logger;
26  
27  import de.ingo.fritzbox.data.Call;
28  import de.ingo.fritzbox.utils.CsvParser;
29  
30  /**
31   * access to call API
32   * @author wf
33   *
34   */
35  public class CallAPIImpl implements CallAPI {
36    // prepare a LOGGER
37    protected static Logger LOGGER = Logger.getLogger("com.bitplan.fritzbox");
38    
39    private FritzBoxSession session;
40  
41    /**
42     * construct me from a FritzBox Session
43     * @param session
44     */
45    public CallAPIImpl(FritzBoxSession session) {
46      this.session = session;
47    }
48  
49    /**
50     * Requests the call list from the FritzBox and returns it.
51     *
52     * @return A list of call objects.
53     * @throws Throwable
54     *           when an exception during communication occurred.
55     */
56    public List<Call> getCallList() throws Throwable {
57      try {
58        
59        final String baseUrl = "/fon_num/foncalls_list.lua";
60        final String params= "?csv=";
61        final String response = session.getResponse(baseUrl, params);
62        return CsvParser.parseCallList(response);
63      } catch (Throwable th) {
64        LOGGER.log(Level.SEVERE, th.getMessage(), th);
65        throw th;
66      }
67    }
68  
69    @Override
70    public void dial(String number) throws Exception {
71      
72      
73    }
74  }

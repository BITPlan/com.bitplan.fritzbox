21  package de.ingo.fritzbox.utils;
22  
23  import java.nio.charset.Charset;
24  import java.text.ParseException;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  import de.ingo.fritzbox.data.Call;
29  
30  
31  /**
32   * A CsvParser to parse csv files provided by the FritzBox.<br>
33   * Currently supported csv files: call list
34   *
35   * @author Ingo Schwarz
36   */
37  public class CsvParser {
38  
39    /**
40     * Parses a call list CSV file.
41     *
42     * @param csvString
43     *            The CSV file as string.
44     * @return The parsed calls as list of Call objects.
45     */
46    public static List<Call> parseCallList(final String csvString) {
47      final List<Call> callerList = new ArrayList<>();
48      String csv = csvString.substring(csvString.indexOf("\n") + 1);
49      csv = csv.substring(csv.indexOf("\n") + 1);
50      csv = new String(csv.getBytes(), Charset.forName("UTF-8"));
51  
52      String line;
53      while (csv.indexOf("\n") > 0 && (line = csv.substring(0, csv.indexOf("\n"))).length() > 6) {
54        csv = csv.substring(csv.indexOf("\n") + 1);
55        callerList.add(parseCallListEntry(line));
56      }
57  
58      return callerList;
59    }
60  
61    /**
62     * Parses a certain line of the call list CSV file.
63     *
64     * @param csvLine
65     *            The line of the CSV call list file.
66     * @return The parsed call.
67     */
68    private static Call parseCallListEntry(final String csvLine) {
69  
70      final String[] lineData = csvLine.split(";");
71      final Call.Builder builder = new Call.Builder().type(lineData[0]).callerName(lineData[2])
72          .callerNumber(lineData[3]).substationName(lineData[4]).substationNumber(lineData[5]);
73  
74      try {
75        builder.date(lineData[1]);
76      } catch (final ParseException e) {
77        // TODO log exception
78      }
79      try {
80        builder.duration(lineData[6]);
81      } catch (final ParseException e) {
82        // TODO log exception
83        e.printStackTrace();
84      }
85  
86      return builder.build();
87    }
88  
89  }
